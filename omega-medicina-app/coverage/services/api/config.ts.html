// Helper to replace path params
export const buildUrl = (endpoint: string, params: Record<string, string> = {}): string => {
  let url = endpoint;
  Object.entries(params).forEach(([key, value]) => {
    url = url.replace(`:${key}`, value);
  });
  return `${API_CONFIG.BASE_URL}${url}`;
};
